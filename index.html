<script>


const SHEET_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vRcmYZt1OfP531U5YVtGh3KWdkg4BaqhWOEv-U8ZnlcPOO73zGDpkO4nbc0geDNVokc7ItMzqSMqGiQ/pub?output=csv"; // <-- EINSETZEN

const LOCATION = "Idar-Oberstein";


let playlist = [];

let index = 0;


// 🔴 CSV laden

async function loadData() {

  try {

    const res = await fetch(SHEET_URL);

    const text = await res.text();


    console.log("RAW CSV:");

    console.log(text);


    const rows = text.split("\n").slice(1);


  playlist = rows.map(row => {


  const cols = row.split(",");


  return {

    location: (cols[0] || "").replaceAll('"','').trim(),

    url: (cols[1] || "").replaceAll('"','').trim(),

    type: (cols[2] || "").replaceAll('"','').trim(),

    duration: parseInt(cols[3]) || 10

  };


}).filter(item =>

  item.location.toLowerCase().includes(LOCATION.toLowerCase())

);


 console.log("ROWS:", rows);

console.log("PLAYLIST FINAL:", playlist);


  } catch (err) {

    console.error("FEHLER BEIM LADEN:", err);

  }

}


// 🔴 Anzeige

function showItem(item) {

  const container = document.getElementById("container");

  container.innerHTML = "";


  if (!item) {

    container.innerHTML = "<h1 style='color:white'>Keine Inhalte</h1>";

    return;

  }


  console.log("SHOW ITEM:", item);


  if (item.type === "image") {

    const img = document.createElement("img");

    img.src = item.url;

    container.appendChild(img);


    setTimeout(next, item.duration * 1000);

  }


  else if (item.type === "video") {

    const video = document.createElement("video");

    video.src = item.url;

    video.autoplay = true;

    video.muted = true;

    video.playsInline = true;


    video.onended = next;


    container.appendChild(video);

  }


  else {

    console.log("UNBEKANNTER TYPE:", item.type);

    setTimeout(next, 3000);

  }

}


// 🔴 nächstes Element

function next() {

  if (playlist.length === 0) {

    console.log("Playlist ist leer!");

    return;

  }


  index = (index + 1) % playlist.length;

  showItem(playlist[index]);

}


// 🔴 Start

async function start() {

  await loadData();


if (playlist.length === 0) {

document.getElementById("container").innerHTML =

"<h1 style='color:white'>❌ Playlist leer – Filter oder CSV Problem</h1>";


console.log("LOCATION FILTER:", LOCATION);

console.log("ROWS:", rows);


return;

}


  showItem(playlist[0]);

}


// 🔴 Refresh alle 60 Sekunden

setInterval(loadData, 60000);


start();


</script>
